        // Gallery Filter

        function filterGallery(category) {

            const items = document.querySelectorAll('.gallery-item');

            const btns = document.querySelectorAll('.filter-btn');

           

            items.forEach(item => {

                if (category === 'all' || item.classList.contains(category)) {

                    item.style.display = 'block';

                } else {

                    item.style.display = 'none';

                }

            });

            btns.forEach(btn => {

                btn.classList.remove('bg-brand-600', 'text-white', 'border-brand-600');

                btn.classList.add('bg-white', 'text-gray-700', 'border-gray-200');

            });

            event.currentTarget.classList.add('bg-brand-600', 'text-white', 'border-brand-600');

            event.currentTarget.classList.remove('bg-white', 'text-gray-700', 'border-gray-200');

        }

        // Form Submission

        const contactForm = document.getElementById('contactForm');

        contactForm.addEventListener('submit', (e) => {

            e.preventDefault();

            contactForm.classList.add('hidden');

            document.getElementById('successMsg').classList.remove('hidden');

        });
